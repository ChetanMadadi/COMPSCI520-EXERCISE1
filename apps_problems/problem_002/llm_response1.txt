def longest_palindromic_substring(s):
    n = len(s)
    if n == 0:
        return 0
    
    max_len = 1
    
    for i in range(n):
        left, right = i, i
        while left >= 0 and right < n and s[left] == s[right]:
            if right - left + 1 > max_len:
                max_len = right - left + 1
            left -= 1
            right += 1
        
        left, right = i, i + 1
        while left >= 0 and right < n and s[left] == s[right]:
            if right - left + 1 > max_len:
                max_len = right - left + 1
            left -= 1
            right += 1
    
    return max_len
